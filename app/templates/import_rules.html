Пример JSON файла

Файл должен содержать массив объектов с полями:

  • name — название правила (обязательно)
  • pattern — регулярное выражение (обязательно)
  • target — цель: url, query, both, headers
  • severity — уровень: critical, high, medium, low
[
    {
        "name": "SQL Injection - Comments",
        "pattern": "(?i)(--|#|/\\*|\\*/)",
        "target": "both",
        "severity": "medium"
    },
    {
        "name": "Block bots",
        "pattern": "(bot|crawler|scraper)",
        "target": "headers",
        "severity": "medium"
    },
    {
        "name": "Custom SQL Injection",
        "pattern": "(?i)(union.*select|select.*from)",
        "target": "both",
        "severity": "high"
    }
]
